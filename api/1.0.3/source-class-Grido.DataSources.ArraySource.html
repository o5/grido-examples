 29:     /** @var array */
 30:     protected $data;
 31: 
 32:     /**
 33:      * @param array $data
 34:      */
 35:     public function __construct(array $data)
 36:     {
 37:         $this->data = $data;
 38:     }

 40:     /**
 41:      * @param Condition $condition
 42:      * @param array $data
 43:      * @return array
 44:      */
 45:     protected function makeWhere(Condition $condition, array $data = NULL)
 46:     {
 47:         $data = $data === NULL
 48:             ? $this->data
 49:             : $data;
 50: 
 51:         $that = $this;
 52:         return array_filter($data, function ($row) use ($condition, $that) {
 53:             if ($condition->callback) {
 54:                 return callback($condition->callback)->invokeArgs(array($condition->value, $row));
 55:             }
 56: 
 57:             $i = 0;
 58:             $results = array();
 59:             foreach ($condition->column as $column) {
 60:                 if (Condition::isOperator($column)) {
 61:                     $results[] = " $column ";
 62: 
 63:                 } else {
 64:                     $i = count($condition->condition) > 1 ? $i : 0;
 65:                     $results[] = (int) $that->compare($row[$column], $condition->condition[$i], $condition->value[$i]);
 66: 
 67:                     $i++;
 68:                 }
 69:             }
 70: 
 71:             $result = implode('', $results);
 72:             return count($condition->column) === 1
 73:                 ? (bool) $result
 74:                 : eval("return $result;");
 75:         });
 76:     }
 77: 
 78:     /**
 79:      * @param string $actual
 80:      * @param string $condition
 81:      * @param mixed $expected
 82:      * @throws \InvalidArgumentException
 83:      * @return bool
 84:      */
 85:     public function compare($actual, $condition, $expected)
 86:     {
 87:         $expected = current((array) $expected);
 88:         $cond = str_replace(' ?', '', $condition);
 89: 
 90:         if ($cond === 'LIKE') {
 91:             $pattern = str_replace('%', '.*', preg_quote($expected));
 92:             return (bool) preg_match("/^{$pattern}$/i", $actual);
 93: 
 94:         } else if ($cond === '=') {
 95:             return $actual == $expected;
 96: 
 97:         } elseif ($cond === 'IS NULL') {
 98:             return $actual === NULL;
 99: 
100:         } elseif ($cond === 'IS NOT NULL') {
101:             return $actual !== NULL;
102: 
103:         } elseif (in_array($cond, array('<', '<=', '>', '>='))) {
104:             return eval("return {$actual} {$cond} {$expected};");
105: 
106:         } else {
107:             throw new \InvalidArgumentException("Condition '$condition' not implemented yet.");
108:         }
109:     }

158:             $data = array();

160:                 $sorter = (string) $item[$column];
161:                 $data[$sorter][] = $item;

165:                 ksort($data);
166:             } else {
167:                 krsort($data);
168:             }
169: 
170:             $this->data = array();

175:             }
176:         }

185:     public function suggest($column, array $conditions, $limit)
186:     {
187:         $data = $this->data;
188:         foreach ($conditions as $condition) {
189:             $data = $this->makeWhere($condition, $data);
190:         }
191:

195:         foreach ($data as $row) {

198:             } elseif (is_callable($column)) {
