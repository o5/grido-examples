 34:         $this->grid = $grid;

 36:             ? ucfirst($this->grid->getName())

 42:     /**
 43:      * @param array $data
 44:      * @param \Nette\ComponentModel\RecursiveComponentIterator $columns
 45:      * @return string
 46:      */
 47:     protected function generateCsv($data, $columns)
 48:     {
 49:         $head = array();
 50:         foreach ($columns as $column) {
 51:             $head[] = $column->getLabel();
 52:         }
 53: 
 54:         $addNewLine = FALSE;
 55:         $source = implode(static::DELIMITER, $head) . static::NEW_LINE;
 56:         foreach ($data as $item) {
 57:             $source .= $addNewLine ? static::NEW_LINE : NULL;
 58: 
 59:             $addDelimiter = FALSE;
 60:             foreach ($columns as $column) {
 61:                 $source .= $addDelimiter ? static::DELIMITER : NULL;
 62:                 $source .= $column->renderExport($item);
 63: 
 64:                 $addDelimiter = TRUE;
 65:             }
 66: 
 67:             $addNewLine = TRUE;
 68:         }
 69: 
 70:         return $source;
 71:     }
 72: 
 73:     /**
 74:      * @internal
 75:      */
 76:     public function handleExport()
 77:     {
 78:         $this->grid->presenter->sendResponse($this);
 79:     }

 88:     {
